cd .. — перейди на уровень выше, в родительскую папку;
cd ~ — перейди в домашнюю директорию (/Users/Username);
cd / — перейди в корневую директорию.

pwd - посмотреть текущую дир

ls - список файлов        флаг -а скрытые файлы

touch - создание файла

mrdir - создание дир      флаг -p создание нескольких дир

cp - что копируем куда копируем

mv - что переносим куда переносим 

cat - прочитать файл

rm - удаление файла       флак -r удаляет все полностью

rmdir - удаление папки

       GIT

1) git init - инициализация репозитория

2) rm -rf .git - разгитить папку

3) git status - состояние репозитория

4) git add файл      флаг --all - добавление файлов для коммита
   git add . - Добавление всей папки

5) git commit -m 'asd' - коммит

6) git log - логи коммитов

7) clip - копирование файла в буфер

8) git remote add - привязка репозитория

9)git push - пуш на гитхаб
 git push -u origin main - в первый раз с укказанием ветки
